vckp_moneta_parse_address_start
CREATE OR REPLACE FUNCTION public.vckp_moneta_parse_address_start(
	vckp_moneta_session_id_in bigint,
	notprocessed integer DEFAULT 1,
	OUT res integer,
	OUT err_mess text,
	OUT action_id_out bigint
)
    RETURNS record
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE SECURITY DEFINER PARALLEL UNSAFE
AS 
$BODY$

DECLARE	
	w_record record;
	actionid bigint;
	r_countall bigint;

BEGIN
	res := 0;
	err_mess:='';
	
	SELECT * FROM f_oleg_getnextactionid (21, vckp_moneta_session_id_in, NULL, 'Обработка') INTO w_record;
	IF w_record.res < 0 THEN
		res := -1;
		err_mess := w_record.err_mess||' ('||w_record.res::text||')';
		RETURN;
	END IF;
	actionid := w_record.action_id;
	action_id_out:=actionid;
	
	SELECT * FROM f_OLEG_listMessagesInsert (actionid, 1, NULL, clock_timestamp()||' - Запуск обработки адресов') INTO w_record;
	SELECT	COUNT(*) INTO r_countall FROM vckp_moneta WHERE vckp_moneta_session_id = vckp_moneta_session_id_in AND (COALESCE(notProcessed, 0) = 0 OR COALESCE(account_new, '') = '');
	SELECT * FROM f_OLEG_listMessagesInsert (actionid, 1, NULL, clock_timestamp()||' - '||r_countall||' записей к обработке') INTO w_record;

	IF COALESCE(notProcessed, 0) = 0 THEN
		UPDATE vckp_moneta
			SET flat_id = NULL,
			    account_new = '',
			    rep_acc = COALESCE(old_rep_acc, rep_acc),
			    house_id = NULL
			WHERE vckp_moneta_session_id = vckp_moneta_session_id_in;
	END IF;

	UPDATE vckp_moneta
		SET res_code = NULL
		WHERE vckp_moneta_session_id = vckp_moneta_session_id_in;

	INSERT INTO vckp_moneta_progress (session_id, total) VALUES (vckp_moneta_session_id_in, r_countall);

	UPDATE vckp_moneta SET period_day=last_day(make_date(SUBSTRING(PERIOD, 1, 4)::int, SUBSTRING(PERIOD, 6, 2)::int,1)) WHERE vckp_moneta_session_id=vckp_moneta_session_id_in;

	TRUNCATE vckp_moneta_tt_parse_address;
	FOR w_record IN SELECT DISTINCT period_day, PERIOD FROM vckp_moneta WHERE vckp_moneta_session_id=vckp_moneta_session_id_in LOOP
		INSERT INTO vckp_moneta_tt_parse_address SELECT w_record.period, * FROM f_oleg_v_house_decision(w_record.period_day) hd WHERE COALESCE(hd.decision_type_id, 0) IN (1, 2, 5);
	END LOOP;

	RETURN;
END;
$BODY$

vckp_moneta_parse_address_final
CREATE OR REPLACE FUNCTION public.vckp_moneta_parse_address_final(
	vckp_moneta_session_id_in bigint,
	action_id_in bigint
)
    RETURNS void
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE SECURITY DEFINER PARALLEL UNSAFE
AS 
$BODY$

DECLARE	
	w_record record;
	actionid bigint;

BEGIN
	actionid:=action_id_in;
	
	UPDATE vckp_moneta vm
		SET	house_id = mf.house_id
		FROM mst_flat mf
		WHERE vm.house_id IS NULL AND
			mf.account = COALESCE(vm.account_new, '') AND
			vm.vckp_moneta_session_id = vckp_moneta_session_id_in;
	
	SELECT * FROM f_OLEG_listMessagesInsert (actionid, 1, NULL, clock_timestamp()||' - разборка адресов завершена') INTO w_record;
	UPDATE vckp_moneta vm
		SET	old_rep_acc = vm.rep_acc,
			rep_acc = vm.account_new,
			flat_id = mf.flat_id,
			sq_pay = round(mf.sqr_base,2),
			els = CASE WHEN COALESCE(els, '') <> '' THEN els ELSE sub.els_id END
		FROM mst_flat mf
			LEFT OUTER JOIN (SELECT flat_id, els_id FROM gis_gkh_ls_now WHERE NOT closed) sub
				ON mf.flat_id = sub.flat_id
		WHERE vckp_moneta_session_id = vckp_moneta_session_id_in AND
			mf.account = COALESCE(vm.account_new,'') AND
			vm.flat_id IS NULL;
			
	UPDATE vckp_moneta vm
		SET res_code = 4
		WHERE vckp_moneta_session_id = vckp_moneta_session_id_in AND
			(
				COALESCE(account_new, '') = '' AND
				COALESCE(house_id, 0) = 0
			);

	UPDATE vckp_moneta vm
		SET res_code = 1
		WHERE vckp_moneta_session_id = vckp_moneta_session_id_in AND
			(
				EXISTS (SELECT 1 FROM mst_flat mf 
							INNER JOIN mst_house mh 
								ON mf.house_id = mh.house_id 
						WHERE mf.account = vm.account_new AND 
							(mf.property_type_id NOT IN (0, 15, 4478535)
							OR 
							mf.flat_owner_id IS NOT NULL 
							OR
							mf.account_closed
							OR
							mh.unfit 
							OR
							mh.billing_locked
							OR 
							NOT EXISTS (SELECT 1 FROM vckp_moneta_tt_parse_address hd WHERE mf.house_id = hd.mst_house_id AND hd.period = vm.period)
							)
					)
			);

	UPDATE vckp_moneta vm
		SET	res_code = 4
		WHERE vckp_moneta_session_id = vckp_moneta_session_id_in AND
			EXISTS (SELECT 1 FROM mst_flat mf
						INNER JOIN mst_house mh
							ON mf.house_id = mh.house_id AND
							   mh.service_system_id = 1
					WHERE mf.account = vm.account_new AND 
						mf.property_type_id = 0);
						
	UPDATE vckp_moneta vm
		SET	res_code = 4
		WHERE vckp_moneta_session_id = vckp_moneta_session_id_in AND
			NOT EXISTS (SELECT 1 FROM vckp_moneta_tt_parse_address hd WHERE COALESCE(vm.house_id, 0) = hd.mst_house_id AND hd.period = vm.period);
						
	UPDATE vckp_moneta vm
		SET res_code = 4
		WHERE vckp_moneta_session_id = vckp_moneta_session_id_in AND
			COALESCE(house_id, 0) > 0 AND
			COALESCE(account_new, '') = '' AND
			res_code IS NULL;
	
							
	SELECT * FROM f_OLEG_listMessagesInsert (actionid, 1, NULL, clock_timestamp()||' - обработка адресов завершена') INTO w_record;	
	TRUNCATE vckp_moneta_progress;
END;
$BODY$
